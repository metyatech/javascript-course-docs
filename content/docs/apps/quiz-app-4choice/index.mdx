4つの選択肢から答えを選ぶクイズアプリを作ります。

このページでは、いきなり完成コードを書くのではなく、次の順番でアプリを大きくしていきます。

1. 1問だけのクイズを作る
2. 問題をデータとしてまとめる
3. 配列から現在の問題を表示する
4. 正解数や回答済み状態を管理する
5. 最後に結果を表示する
6. 自分のクイズに改造する

---

```js
const questions = [
    {
        text: 'JavaScriptでクリックイベントを登録するメソッドは？',
        choices: ['querySelector', 'addEventListener', 'textContent', 'console.log'],
        answer: 'addEventListener'
    },
    {
        text: '配列の最初の要素を取り出す番号は？',
        choices: ['0', '1', '2', '3'],
        answer: '0'
    },
    {
        text: '要素の文字を変更するプロパティは？',
        choices: ['value', 'length', 'textContent', 'styleSheet'],
        answer: 'textContent'
    }
];

let currentIndex = 0;
let score = 0;
let answered = false;

const progressText = document.querySelector('#progressText');
const questionText = document.querySelector('#questionText');
const messageText = document.querySelector('#messageText');
const nextButton = document.querySelector('#nextButton');

const choice1 = document.querySelector('#choice1');
const choice2 = document.querySelector('#choice2');
const choice3 = document.querySelector('#choice3');
const choice4 = document.querySelector('#choice4');

const choiceButtons = [choice1, choice2, choice3, choice4];

function showQuestion() {
    const question = questions[currentIndex];

    progressText.textContent = currentIndex + 1 + '問目 / ' + questions.length + '問中';
    questionText.textContent = question.text;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].textContent = question.choices[i];
        choiceButtons[i].disabled = false;
    }

    messageText.textContent = '選択肢を選んでください。';
    messageText.className = '';
    answered = false;
}

function checkAnswer(selectedAnswer) {
    if (answered === true) {
        return;
    }

    const question = questions[currentIndex];

    if (selectedAnswer === question.answer) {
        score++;
        messageText.textContent = '正解！';
        messageText.className = 'correct';
    } else {
        messageText.textContent = '不正解。正解は「' + question.answer + '」です。';
        messageText.className = 'wrong';
    }

    answered = true;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].disabled = true;
    }
}

function showFinalResult() {
    progressText.textContent = '終了';
    questionText.textContent = '結果は ' + questions.length + '問中 ' + score + '問正解です。';
    messageText.textContent = 'おつかれさまでした。';

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].textContent = '-';
        choiceButtons[i].disabled = true;
    }

    nextButton.disabled = true;
}

choice1.addEventListener('click', () => {
    checkAnswer(choice1.textContent);
});

choice2.addEventListener('click', () => {
    checkAnswer(choice2.textContent);
});

choice3.addEventListener('click', () => {
    checkAnswer(choice3.textContent);
});

choice4.addEventListener('click', () => {
    checkAnswer(choice4.textContent);
});

nextButton.addEventListener('click', () => {
    if (answered === false) {
        messageText.textContent = '先に選択肢を選んでください。';
        return;
    }

    currentIndex++;

    if (currentIndex < questions.length) {
        showQuestion();
    } else {
        showFinalResult();
    }
});

showQuestion();
```

<Action title="showQuestion を書く">

```js
function showQuestion() {

    progressText.textContent = currentIndex + 1 + '問目 / ' + questions.length + '問中';
    questionText.textContent = question.text;

    for (let i = 0; i < choiceButtons.length; i++) {
```

<Concept title="for 文の境界条件">

そのため、条件は次のようにします。

```js
i < choiceButtons.length
```

`i <= choiceButtons.length` にすると、最後に存在しない5個目を取り出そうとしてエラーになります。

</Concept>

<Section title="バグ修正チャレンジ" goal="状態管理で起きやすいミスを見つけます。">

次のコードには、よくあるミスが6つ入っています。  
`id`、配列番号、比較演算子、関数呼び出し、最終結果の観点で探してください。

<Exercise>

```js
let currentIndex = 1;
let score = 0;
let answered = false;

const progressText = document.querySelector('#progress');
const questionText = document.querySelector('#questionText');

function showQuestion() {
    const question = questions[currentIndex];

    progressText.textContent = currentIndex + 1 + '問目';
    questionText.textContent = question.text;

    for (let i = 0; i <= choiceButtons.length; i++) {
        choiceButtons[i].textContent = question.choices[i];
    }

    answered = false;
}

function checkAnswer(selectedAnswer) {
    if (answered = true) {
        return;
    }

    const question = questions[currentIndex];

    if (selectedAnswer === question.answer) {
        score++;
    }

    answered = true;
}

nextButton.addEventListener('click', () => {
    currentIndex++;

    if (currentIndex < questions.length) {
        showQuestion;
    }
});
```

<Solution>

1. `currentIndex = 1` では2問目から始まるため、`0` にする。
2. `#progress` はHTML側の `id="progressText"` と一致しないため、`#progressText` にする。
3. `i <= choiceButtons.length` は配列範囲外に進むため、`i < choiceButtons.length` にする。
4. `if (answered = true)` は代入なので、`if (answered === true)` にする。
5. `showQuestion;` は関数を呼んでいないため、`showQuestion();` にする。
6. 最後の問題を終えたときの `else { showFinalResult(); }` がないため、追加する。

</Solution>

</Exercise>

</Section>
