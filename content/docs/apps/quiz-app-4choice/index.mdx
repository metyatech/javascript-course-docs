import Exercise, { QuickCheck, Hint, Answer } from '@metyatech/exercise/client';

4つの選択肢から答えを選び、最後に正解数が表示されるクイズアプリを作ります。

このページでは、HTMLとCSSを用意したあと、JavaScriptを次の順番で組み立てます。

1. 1問だけの正解判定を作る
2. 問題をデータとしてまとめる
3. 問題番号、正解数、回答済みかどうかを変数で覚える
4. 現在の問題を画面に表示する
5. 回答を判定して、次の問題へ進める
6. 最後に結果を表示し、もう一度遊べるようにする
7. 自分のテーマのクイズに改造する

演習1〜4が基本課題です。`演習-発展` は、基本課題を完成させた人だけ取り組みます。

```js
let questions = [
    {
        text: '次のうち、HTML要素にイベント処理を登録するメソッドは？',
        choices: ['querySelector', 'addEventListener', 'textContent', 'console.log'],
        answerIndex: 1
    },
    {
        text: '配列の最初の要素を取り出すインデックスは？',
        choices: ['0', '1', '2', '3'],
        answerIndex: 0
    },
    {
        text: '次のうち、HTML要素の文字を取得・設定できるプロパティは？',
        choices: ['value', 'length', 'textContent', 'querySelector'],
        answerIndex: 2
    }
];

let currentIndex = 0;
let score = 0;
let answered = false;

let progressText = document.querySelector('#progressText');
let questionText = document.querySelector('#questionText');
let choices = document.querySelector('#choices');
let messageText = document.querySelector('#messageText');
let nextButton = document.querySelector('#nextButton');
let restartButton = document.querySelector('#restartButton');

let choice1 = document.querySelector('#choice1');
let choice2 = document.querySelector('#choice2');
let choice3 = document.querySelector('#choice3');
let choice4 = document.querySelector('#choice4');

let choiceButtons = [choice1, choice2, choice3, choice4];

function showQuestion() {
    let question = questions[currentIndex];

    progressText.textContent = currentIndex + 1 + '問目 / ' + questions.length + '問中';
    questionText.textContent = question.text;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].textContent = question.choices[i];
        choiceButtons[i].disabled = false;
        choiceButtons[i].className = '';
    }

    choices.style.display = 'block';
    messageText.textContent = '選択肢を選んでください。';
    messageText.className = '';
    nextButton.style.display = 'block';
    restartButton.style.display = 'none';
    answered = false;
}

function checkAnswer(selectedIndex) {
    if (answered === true) {
        return;
    }

    let question = questions[currentIndex];
    answered = true;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].disabled = true;
    }

    choiceButtons[question.answerIndex].className = 'correct';

    if (selectedIndex === question.answerIndex) {
        score++;
        messageText.textContent = '正解！';
        messageText.className = 'correct';
    } else {
        choiceButtons[selectedIndex].className = 'wrong';
        let correctAnswer = question.choices[question.answerIndex];
        messageText.textContent = '不正解。正解は「' + correctAnswer + '」です。';
        messageText.className = 'wrong';
    }
}

function showFinalResult() {
    progressText.textContent = '終了';
    questionText.textContent = '結果は ' + questions.length + '問中 ' + score + '問正解です。';
    choices.style.display = 'none';
    messageText.textContent = 'おつかれさまでした。';
    messageText.className = '';
    nextButton.style.display = 'none';
    restartButton.style.display = 'block';
}

choice1.addEventListener('click', function () {
    checkAnswer(0);
});

choice2.addEventListener('click', function () {
    checkAnswer(1);
});

choice3.addEventListener('click', function () {
    checkAnswer(2);
});

choice4.addEventListener('click', function () {
    checkAnswer(3);
});

nextButton.addEventListener('click', function () {
    if (answered === false) {
        messageText.textContent = '先に選択肢を選んでください。';
        return;
    }

    currentIndex++;

    if (currentIndex < questions.length) {
        showQuestion();
    } else {
        showFinalResult();
    }
});

restartButton.addEventListener('click', function () {
    currentIndex = 0;
    score = 0;
    showQuestion();
});

showQuestion();
```

<Section title="現在の問題を画面に表示する" goal="currentIndexに対応する問題文と4つの選択肢を画面へ表示します。">

`questions[currentIndex]` で、今表示する問題のオブジェクトを取り出します。

たとえば `currentIndex` が `0` のとき、次のように値をたどります。

| コード | 取り出す値 |
| --- | --- |
| `questions[0]` | 1問目のオブジェクト |
| `questions[0].text` | 1問目の問題文 |
| `questions[0].choices[2]` | 1問目の3番目の選択肢 |

4つのボタンへ選択肢を入れるときは、同じ `i` を2つの配列に使います。

| `i` | 書き換えるボタン | 入れる選択肢 |
| ---: | --- | --- |
| `0` | `choiceButtons[0]` | `question.choices[0]` |
| `1` | `choiceButtons[1]` | `question.choices[1]` |
| `2` | `choiceButtons[2]` | `question.choices[2]` |
| `3` | `choiceButtons[3]` | `question.choices[3]` |

ボタン要素の `disabled` に `false` を入れると、ボタンを押せる状態になります。  
要素の `className` に空文字 `''` を入れると、前の問題で付けた `correct` または `wrong` というクラス名を取り除けます。

JavaScriptからCSSを変更するときは、既習の `要素.style.CSSプロパティ名 = '値'` を使います。  
ここでは `style.display` を使って、選択肢とボタンの表示・非表示を切り替えます。

<Action title="showQuestion関数を書く">

要素取得のコードの下に、次の関数を追加します。

```js
function showQuestion() {
    let question = questions[currentIndex];

    progressText.textContent = currentIndex + 1 + '問目 / ' + questions.length + '問中';
    questionText.textContent = question.text;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].textContent = question.choices[i];
        choiceButtons[i].disabled = false;
        choiceButtons[i].className = '';
    }

    choices.style.display = 'block';
    messageText.textContent = '選択肢を選んでください。';
    messageText.className = '';
    nextButton.style.display = 'block';
    restartButton.style.display = 'none';
    answered = false;
}
```

`display` に `'none'` を入れると非表示になります。このアプリでは、選択肢エリアと操作ボタンをブロック表示にしているため、再表示するときは `'block'` を入れます。

</Action>

<Action title="最初の問題を表示する">

```js

```

<Section title="選ばれた答えを判定する" goal="選ばれた位置と正解の位置を比べ、得点と画面表示を更新します。">

`checkAnswer` 関数には、押された選択肢の位置を `selectedIndex` として渡します。

- 1番目のボタンなら `0`
- 2番目のボタンなら `1`
- 3番目のボタンなら `2`
- 4番目のボタンなら `3`

現在の問題が持っている `answerIndex` と `selectedIndex` が同じなら正解です。

関数の途中で `return` を実行すると、その時点で関数の処理が終了します。すでに回答済みなら、得点や表示を変更せずに終了させます。

<Action title="checkAnswer関数を書く">

`showQuestion` 関数の下に、次の関数を追加します。

```js
function checkAnswer(selectedIndex) {
    if (answered === true) {
        return;
    }

    let question = questions[currentIndex];
    answered = true;

    for (let i = 0; i < choiceButtons.length; i++) {
        choiceButtons[i].disabled = true;
    }

    choiceButtons[question.answerIndex].className = 'correct';

    if (selectedIndex === question.answerIndex) {
        score++;
        messageText.textContent = '正解！';
        messageText.className = 'correct';
    } else {
        choiceButtons[selectedIndex].className = 'wrong';
        let correctAnswer = question.choices[question.answerIndex];
        messageText.textContent = '不正解。正解は「' + correctAnswer + '」です。';
        messageText.className = 'wrong';
    }
}
```

回答後は、すべてのボタンを `disabled = true` にして押せない状態にします。

最初の `if (answered === true)` は追加の安全対策です。画面上ではボタンを無効にして二重回答を防いでいますが、何らかの理由で `checkAnswer` がもう一度呼ばれても、`score` が増えないようにしています。

</Action>

<Action title="4つのボタンと判定関数をつなぐ">

`script.js` の末尾にある `showQuestion();` より上へ、次のイベント処理を追加します。

    checkAnswer(1);
});

    checkAnswer(2);
});

    checkAnswer(3);
});
```

</Action>

<Section title="次の問題と結果画面を作る" goal="回答後に次の問題へ進み、最後は結果と再スタートボタンを表示します。">

<Action title="結果を表示する関数を書く">

    questionText.textContent = '結果は ' + questions.length + '問中 ' + score + '問正解です。';
    choices.style.display = 'none';

    messageText.className = '';
    nextButton.style.display = 'none';
    restartButton.style.display = 'block';
}

<Action title="次の問題へ進む処理を書く">

nextButton.addEventListener('click', function () {

### 演習-発展3

<Exercise>

1問目から始まること、正しい要素を取得すること、4つの選択肢を表示すること、回答済みかを比較すること、次の問題を表示することの観点で確認します。

    progressText.textContent = currentIndex + 1 + '問目';

    for (let i = 0; i <= choiceButtons.length; i++) {

    if (answered = true) {
        return;
    }

    let question = questions[currentIndex];

    if (selectedIndex === question.answerIndex) {
        score++;
    }

    answered = true;
}

    if (answered === false) {
        return;

    currentIndex++;

    if (currentIndex < questions.length) {
        showQuestion;
    } else {
        showFinalResult();
    }
});
```
