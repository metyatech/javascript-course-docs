## ヒット＆ブローって何？

ヒット＆ブローは、0〜9の数字を4つ使い、同じ数字をくり返さない並びを当てるゲームです。予想を入力すると、コンピュータがヒットとブローで返事をします。

- ヒット：数字も場所も合っている個数
- ブロー：数字は合っているが、場所が違う個数

正解が `1234` のときは、次のように判定します。

| 予想 | 返事 | 意味 |
| :--: | :-- | :-- |
| `5678` | `0ヒット 0ブロー` | 使われている数字がありません |
| `1256` | `2ヒット 0ブロー` | `1` と `2` が同じ場所にあります |
| `4321` | `0ヒット 4ブロー` | 4つとも数字は合っていますが、場所が違います |
| `1234` | `4ヒット 0ブロー` | クリアです |

### 演習1

<Exercise>

まずは、プレイヤーの予想を受け取れるようにしましょう。正解は今だけ `let secret = '1234';` と固定しておきます。

`prompt` で4文字の入力を受け取り、ちょうど4文字で、すべて `0`〜`9` なら `OK`、そうでなければ `invalid` と出力してください。テストしやすいように、今は正解も表示しておきましょう。

<Solution>
<CodePreview>
```javascript
    let secret = '1234';
    console.log('正解: ' + secret); // 出力: 正解: 1234
    let guess = prompt('4桁の数字を入力してね');
    let valid = guess.length === 4;

    for (let i = 0; i < 4; i++) {
        if (guess[i] < '0' || guess[i] > '9') valid = false;
    }

    if (valid === true) {
        console.log('OK'); // 出力: OK
    } else {
        console.log('invalid'); // 出力: invalid
    }
```

演習1のコードに、同じ数字が2か所にないかを調べる入れ子の `for` を追加してください。重複があれば `重複` と出力します。

### 演習4

<Exercise>

ヒットが少ない問題は簡単です。ブローは別の桁を見る必要があるので、`for` を入れ子にしましょう。

`guess` の数字が `secret` のどこかにあるかを全部の組み合わせで見ます。ただし同じ場所ならヒットなので、ブローには入れません。

<Solution>
<CodePreview>
```javascript
    let secret = '1234';
    console.log('正解: ' + secret); // 出力: 正解: 1234
    let guess = prompt('4桁の数字を入力してね');
    let valid = guess.length === 4;
    let duplicate = false;

    for (let i = 0; i < 4; i++) {
        if (guess[i] < '0' || guess[i] > '9') valid = false;
        for (let j = i + 1; j < 4; j++) {
            if (guess[i] === guess[j]) duplicate = true;
        }
```

### 演習6

<Exercise>

10回まで挑戦できるようにして、当たったら抜けられるようにしましょう。

演習5までの入力、チェック、ヒット・ブロー計算を `for (let attempt = 1; attempt <= 10; attempt++)` で囲みます。ヒットが4になったら `break` でループを抜け、ループのあとで成功か失敗を出力してください。

<Solution>
<CodePreview>
```javascript
    let secret = '1234';
    console.log('正解: ' + secret); // 出力: 正解: 1234
    let hit = 0;
    let clearAttempt = 0;

    for (let attempt = 1; attempt <= 10; attempt++) {
        let guess = prompt(attempt + '回目：4桁の数字を入力してね');
        let valid = guess.length === 4;
        let duplicate = false;

        for (let i = 0; i < 4; i++) {
            if (guess[i] < '0' || guess[i] > '9') valid = false;
            for (let j = i + 1; j < 4; j++) {
                if (guess[i] === guess[j]) duplicate = true;
            }
        }

        if (valid === false) console.log('invalid'); // 出力: invalid
        else if (duplicate === true) console.log('重複'); // 出力: 重複
        else {
            hit = 0;
            let blow = 0;
            for (let i = 0; i < 4; i++) {
                if (secret[i] === guess[i]) hit = hit + 1;
                for (let j = 0; j < 4; j++) {
                    if (i !== j && guess[i] === secret[j]) blow = blow + 1;
                }
            }
            console.log(attempt + '回目: ヒント: ヒット ' + hit + '、ブロー ' + blow);
            if (hit === 4) {
                clearAttempt = attempt;
                break;
            }
        }
    }

    if (hit === 4) {
        console.log('クリア！' + clearAttempt + '回で当てた！');
    } else {
        console.log('残念！正解は ' + secret + ' でした');
    }

    // 成功例（6回目で当たり）
    // 出力: 1回目: ヒント: ヒット 0、ブロー 0
    // 出力: 2回目: ヒント: ヒット 1、ブロー 1
    // 出力: 3回目: ヒント: ヒット 2、ブロー 0
    // 出力: 4回目: ヒント: ヒット 0、ブロー 4
    // 出力: 5回目: ヒント: ヒット 3、ブロー 0
    // 出力: 6回目: ヒント: ヒット 4、ブロー 0
    // 出力: クリア！6回で当てた！
    // 失敗例（10回以内に当たらなかった場合）
    // 出力: 10回目: ヒント: ヒット 1、ブロー 2
    // 出力: 残念！正解は 1234 でした
```

### 演習-発展2

<Exercise>

ゲームの最初に難しさを選べるようにしましょう。`3` / `4` / `5` のどれかを入力し、その桁数で最後まで判定します。

数字の `4` と書いていた場所のうち、桁数を表すものを `digitCount` に変えます。入力の文字数チェック、ヒットとブローの `for` の範囲、クリア判定の `hit === 4` を置きかえましょう。

<Solution>
<CodePreview>
```javascript
    let digitCount = Number(prompt('難易度を選んでね（3 / 4 / 5）'));
    if (digitCount !== 3 && digitCount !== 4 && digitCount !== 5) digitCount = 4;
    let secret = '123';
    if (digitCount === 4) secret = '1234';
    else if (digitCount === 5) secret = '12345';
    let hit = 0;
    let clearAttempt = 0;

    for (let attempt = 1; attempt <= 10; attempt++) {
        let guess = prompt(attempt + '回目：' + digitCount + '桁の数字を入力してね');
        let valid = guess.length === digitCount;
        let duplicate = false;

        for (let i = 0; i < digitCount; i++) {
            if (guess[i] < '0' || guess[i] > '9') valid = false;
            for (let j = i + 1; j < digitCount; j++) {
                if (guess[i] === guess[j]) duplicate = true;
            }
        }

        if (valid === false) console.log('invalid'); // 出力: invalid
        else if (duplicate === true) console.log('重複'); // 出力: 重複
        else {
            hit = 0;
            let blow = 0;
            for (let i = 0; i < digitCount; i++) {
                if (secret[i] === guess[i]) hit = hit + 1;
                for (let j = 0; j < digitCount; j++) {
                    if (i !== j && guess[i] === secret[j]) blow = blow + 1;
                }
            }
            console.log(attempt + '回目: ヒント: ヒット ' + hit + '、ブロー ' + blow);
            if (hit === digitCount) {
                clearAttempt = attempt;
                break;
            } else {
                let remaining = 10 - attempt;
                console.log('あと ' + remaining + ' 回');
                if (remaining < 3) console.log('もうすぐです！');
            }
        }
    }

    if (hit === digitCount) {
        console.log('クリア！' + clearAttempt + '回で当てた！');
    } else {
        console.log('残念！正解は ' + secret + ' でした');
    }
```
